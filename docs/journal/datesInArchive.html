Provider<List<DateTime>> datesInArchive
final

Dates at which there are exist archived services (aggregate from controllerDatesInArchive).

{@category Providers}

Implementation

final datesInArchive = Provider<List<DateTime>>((ref) {
  return <DateTime>{
    ...ref
        .watch(workerProfiles)
        .map((e) => ref.watch(controllerDatesInArchive(e.apiKey)))
        .reduce((value, element) => value.addAll(element) as List<DateTime>),
  }.toList();
});
