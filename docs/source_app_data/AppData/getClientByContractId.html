ClientProfile? getClientByContractId(WorkerProfile workerProfile, int? cId) {
  if (workerProfile.clients.isNotEmpty) {
    for (final client in workerProfile.clients) {
      if (client.contractId == cId) {
        return client;
      }
    }
  }

  return null;
}
